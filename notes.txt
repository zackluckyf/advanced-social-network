'use strict';

var SEQUELIZE = require('sequelize'),
    sequelize = new SEQUELIZE('...', '...', '...', {
        host: '...',
        dialect: 'postgres'
    });

var testModel = sequelize.define('testModel', {
    title     : SEQUELIZE.TEXT,
    id_list  : { type : SEQUELIZE.ARRAY(SEQUELIZE.INTEGER), defaultValue: null}
}, {
    tableName : 'test',
    createdAt : false,
    updatedAt : false,
    deletedAt : false
});

testModel.create({title : 'testing 2', id_list: [1, 2]});
